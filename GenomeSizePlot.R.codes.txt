###Genome size plotting # load the data into RGenome_size_plot_data <- readxl::read_excel("TE_families_Species_Genome_for_Linear.xlsx", sheet = "Sheet3")# Converting data from wide to long formatdata_long <- Genome_size_plot_data                 data_long$subgroup <- as.factor(1:nrow(data_long))# Stacked barplot using ggplot2library(ggplot2)ggplot(data_long,                        aes(x = Species,           y = Genomesize))+  geom_bar(stat = "identity", fill = "turquoise4")
