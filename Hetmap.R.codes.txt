##TE heatmap for PPN_TE paperrequire("RColorBrewer")mycol <- colorRampPalette(c("green", "darkgreen"))(100)myBreaks <- seq(0, 6, length.out=100)library("pheatmap")test2 <- read.table("TE_data_log_transformed_for_heatmap.txt", header = T)test2[test2==0] <-NA#test2 <- na.omit(test2)rownames(test2)=test2[,1]#rownames(annotation_c) <- colnames(test2)chars=test2[,c(1,2)]test3=test2[,c(2:27)]pheatmap(as.matrix(test3), scale = "none", clustering_distance_rows = "correlation",          clustering_method = "complete",color =mycol,  fontsize_col=10,          fontsize_row = 7,annotation_row = chars[1], cluster_rows = F,          cluster_cols = F, na_col = "white", breaks = myBreaks, annotation_names_row = F, annotation_names_col =F)
